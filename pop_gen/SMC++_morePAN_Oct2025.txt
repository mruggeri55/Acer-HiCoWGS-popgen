# make bed file for uncalled loci, filtered snps, and repetitive regions
# convert good sites file (sites_rmHWE.txt see diversity stats) to bed file
# then all missing sites should be masked
sites=/scratch1/mruggeri/AcerHapPanel/pop_gen/pixy/sites_rmHWE.txt
awk 'BEGIN{OFS="\t"} {print $1, ($3 = $2 - 1), $2}' $sites > good_sites_snps_and_invariant.bed

salloc --partition=meb_largemem --time=3:00:00
module load bedtools2/2.31.1
gBED=/project2/ckenkel_26/RefSeqs/AcerGenome/Sanger_Inst_GCA_964034985.1/GCA_964034985.1_jaAcrCerv1.1_genomic.fna.fai
bedtools merge -i good_sites_snps_and_invariant.bed > temp.bed
bedtools sort -g $gBED -i temp.bed > temp2.bed
bedtools complement -i temp2.bed -g $gBED > mask.bed

# then need to combine repeat mask bed and filtered snp bed
gREPS=/scratch1/mruggeri/AcerHapPanel/vcfSanger/final_merge/Acer_repeats.bed
cat $gREPS mask.bed > SMCmask.bed
bedtools sort -g $gBED -i SMCmask.bed | bedtools merge -i stdin > SMCmask_sorted_merged.bed
bgzip SMCmask_sorted_merged.bed
tabix -p bed SMCmask_sorted_merged.bed.gz 

# now can move onto SMC++
salloc --partition=meb_largemem --ntasks=30 --time=3:00:00 --mem=0
# run this for each population -- change POP list and prefix for naming
MASK=/scratch1/mruggeri/AcerHapPanel/pop_gen/SMCpp_morePAN/SMCmask_sorted_merged.bed.gz 
VCF=/project2/ckenkel_26/Acer_WGS/vcfs/Sep2025_morePAN/snps.5bpind.merged.morePAN.minDP10.maxDP50.qual.SOR.miss10.vcf.gz
SIF=/scratch1/mruggeri/AcerHapPanel/pop_gen/SMCpp/smcpp_latest.sif
lists=/scratch1/mruggeri/AcerHapPanel/pop_gen/lists
CHROMS=$lists/chrom.list
POP=$lists/AR_CU.pop.list  
prefix=AR_CU
module load apptainer

# subset samples and chromosome from vcf and convert to smc format
mkdir $prefix'_vcf2smc'
for CHROM in `cat $CHROMS`; do
for SAMPLE in `cat $POP`; do
apptainer exec --bind $PWD,/scratch1/mruggeri,/project2/ckenkel_26 $SIF \
 smc++ vcf2smc -d $SAMPLE $SAMPLE --mask $MASK $VCF $prefix'_vcf2smc'/$prefix'_'$SAMPLE'_'$CHROM'.smc.gz' $CHROM $prefix':'$(echo $SAMPLE);
 done; done

# estimate Ne
# you can do this for one chrom with one core but all requires multiple

--cores 30 --em-iterations 50 --timepoints 20 200000 --thinning 3000 --knots 20 4e-9 \
$prefix'_vcf2smc'/*.smc.gz -o $prefix'_out'
